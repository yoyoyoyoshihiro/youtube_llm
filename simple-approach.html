ブックマークレット:
以下のコードをブラウザのブックマークに保存すると、 YouTubeページで直接NotebookLMを開けます
javascript:(function(){const url=window.location.href;if(url.includes('youtube.com')||url.includes('youtu.be')){window.open('https://notebooklm.google.com/','_blank');navigator.clipboard.writeText(url).then(()=>alert('URLをコピーしました！NotebookLMに貼り付けてください'));}else{alert('YouTubeページで実行してください');}})();
